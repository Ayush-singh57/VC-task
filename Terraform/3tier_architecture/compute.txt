# Find the latest Ubuntu 22.04 AMI automatically
data "aws_ami" "ubuntu" {
  most_recent = true
  owners      = ["099720109477"] 
  filter {
    name   = "name"
    values = ["ubuntu/images/hvm-ssd/ubuntu-jammy-22.04-amd64-server-*"]
  }
}

# 1. The Web Server (Public)
resource "aws_instance" "web_server" {
  ami                    = data.aws_ami.ubuntu.id
  instance_type          = "t2.micro"
  subnet_id              = aws_subnet.web_subnet.id
  vpc_security_group_ids = [aws_security_group.web_sg.id]
  # key_name             = "your-key-name" # Uncomment and add your AWS Key Pair name if you want to SSH in

  tags = { Name = "Tier-1-Web-Server" }
}

# 2. The App Server (Private)
resource "aws_instance" "app_server" {
  ami                    = data.aws_ami.ubuntu.id
  instance_type          = "t2.micro"
  subnet_id              = aws_subnet.app_subnet.id
  vpc_security_group_ids = [aws_security_group.app_sg.id]
  # key_name             = "your-key-name" 

  tags = { Name = "Tier-2-App-Server" }
}

# 3. The Database Server (Private)
resource "aws_instance" "db_server" {
  ami                    = data.aws_ami.ubuntu.id
  instance_type          = "t2.micro"
  subnet_id              = aws_subnet.db_subnet.id
  vpc_security_group_ids = [aws_security_group.db_sg.id]
  # key_name             = "your-key-name" 

  tags = { Name = "Tier-3-DB-Server" }
}

# Output the one public IP we have
output "web_public_ip" {
  description = "The public IP of your Web Server (The Front Door)"
  value       = aws_instance.web_server.public_ip
}
